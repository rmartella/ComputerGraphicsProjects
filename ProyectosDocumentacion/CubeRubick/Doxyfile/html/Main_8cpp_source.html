136 void GLApplication::ApplicationLoop() {
137  // Se crea el rubick con su modelo que corresponde a un cubo.
138  Node rootNode = Node(&g_cube, false);
139  // Se crean los nodos y la estructura gerarquica del cubo de rubick.
140  RubiksCube rubikCube = RubiksCube(&rootNode);
141  // Se le indica al rubick sobre que objeto se estara logeando los movimientos.
142  rubikCube.SetRotationCubeManager(this->rotationCubeManager);
143  // Declaración de variable para incrementar su rotación.

146

148  bool completeFaceInit = false;
149

152  TimeManager::Instance().CalculateFrameRate(false);
153 
154  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
155  glClearColor(0.0f, 0.0f, 0.4f, 0.0f);
156 
157  glm::mat4 parentMatrix(1.0f);
158  glm::mat4 rotateRoot(glm::toMat4(gRotation));
159  parentMatrix = glm::translate(parentMatrix, vec3(0, 0, -8));
160  parentMatrix = parentMatrix * rotateRoot;
161  rootNode.setTransformMatrix(&parentMatrix);
162  // Se le settea al manejador de la orientación la matriz asociada al cubo.
163  this->rotationCubeManager->SetWorldMatrix(parentMatrix);
164 
165  // Se incrmenta la rotación.
166  ratio1 = ratio1 + TimeManager::Instance().DeltaTime * speedRotattionFaces;
167 
168  // Inicialización de los movimientos, de la ultima sesión.
169  if (!completeFaceInit) {
170  if (this->rotationCubeManager->GetMovimientosSave()->size() > 0) {
171  rubikCube.rotateFaceAuto(fmod(ratio1, 1), false, &completeFaceInit);
172  } else
173  completeFaceInit = true;
174  if (completeFaceInit)
175  speedRotattionFaces = 1.0f;
176  } else {
177  // Se rota la cara solicitada por el usuario.
178  if (!readyReset && this->rotationCubeManager->getCurrentRotation() != NULL)
179  rubikCube.rotateFace(fmod(ratio1, 1), armar, &readyReset);
180  else if (!readyReset && !armar)
181  ratio1 = 0;
182  else if (armar && !completeFaceAuto
183  && this->rotationCubeManager->GetMovimientosSave()->size() > 0) {
184  // Se arma el cubo de rubick.
185  readyReset = true;
186  speedRotattionFaces = 2.0f;
187  rubikCube.rotateFaceAuto(fmod(ratio1, 1), armar, &completeFaceAuto);
188  } else if (completeFaceAuto) {
189  // Se completa la rotación automatica del reinicio del estado.
190  readyReset = false;
191  completeFaceAuto = false;
192  armar = false;
193  ratio1 = 0;
194  speedRotattionFaces = 1.0f;
195  while (this->rotationCubeManager->GetMovimientosSave()->size() > 0) {
196  this->rotationCubeManager->GetMovimientosSave()->pop_back();
197  }
198  }
199  }
