156 bool RotationCubeManager::RayTriangleIntersect(glm::vec3 orig, glm::vec3 dir,
157  glm::vec3 vert0, glm::vec3 vert1,
158  glm::vec3 vert2, double *t,
159  double *u, double *v) {
160  glm::vec3 edge1, edge2, tvec, pvec, qvec;
161  double det, inv_det;
162  // Se encuentran los vectores de dos aristas que contengan a ver0
163  edge1 = glm::vec3(vert1 - vert0);
164  edge2 = glm::vec3(vert2 - vert0);
165  // Se calcula el determiante - se usa para calcular el parametro U.
166  pvec = glm::cross(dir, edge2);
167  // Si el determinante es muy cercano a cero, rayo cae en el plano
168  // que contiene al triángulo.
169  det = glm::dot(edge1, pvec);
170  if (det > -EPSILON && det < EPSILON)
171  return false;
172  inv_det = 1.0 / det;
173  // Distancia entre el vert0 al origen del rayo.
174  tvec = glm::vec3(orig - vert0);
175  // Calcula el parametro u y prueba su valor.
176  *u = glm::dot(tvec, pvec) * inv_det;
177  if (*u < 0.0 || *u > 1.0)
178  return false;
179  qvec = glm::cross(tvec, edge1);
180  // Calcula el parametro v y prueba que este dentro del rango.
181  *v = glm::dot(dir, qvec) * inv_det;
182  if (*v < 0.0 || *u + *v > 1.0)
183  return false;
184  // Calcula t, interseccion del rayo con el triángulo
185  *t = glm::dot(edge2, qvec) * inv_det;
186  return true;
187 }
