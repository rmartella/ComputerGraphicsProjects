26 bool Texture::Load() {
27  // Se obtiene del nombre del archivo a cargar en la GPU como un apuntador a un arreglo de chars
28  const char* filename = m_fileName.c_str();
29 
30  // Determina el formato de la imagen
31  FREE_IMAGE_FORMAT format = FreeImage_GetFileType(filename, 0);
32 
33  // Si la imagen no es encontrada termina el programa
34  if (format == -1) {
35  std::cout << "No se encontró la imagen: " << m_fileName << " - Abortando."
36  << std::endl;
37  exit(-1);
38  }
39  // Si el formato no es soportado por FreeImage termina el programa
40  if (format == FIF_UNKNOWN) {
41  std::cout
42  << "No se puede determinar el formato de imagen - validarla extensión del archivo..."
43  << std::endl;
44 
45  // Se obtiene el formato del archivo
46  format = FreeImage_GetFIFFromFilename(filename);
47 
48  // Revisa si la librería es capaz de leer el formato
49  if (!FreeImage_FIFSupportsReading(format)) {
50  std::cout << "El formato de la imagen no puede ser leeído!" << std::endl;
51  exit(-1);
52  }
53  }
54  // Si es valida la imagen y puede ser leeído, se carga la imagen en un bitap
55  FIBITMAP* bitmap = FreeImage_Load(format, filename);
56  FreeImage_FlipVertical(bitmap);
57 
58  // Obtiene el número de bits por pixel de la imagen
59  int bitsPerPixel = FreeImage_GetBPP(bitmap);
60 
61  // Convierte la imagen a 32 bits (8 bits por canal).
62  FIBITMAP* bitmap32;
63  if (bitsPerPixel == 32) {
64  /*std::cout << "Source image has " << bitsPerPixel
65  << " bits per pixel. Skipping conversion." << std::endl;*/
66  bitmap32 = bitmap;
67  } else {
68  /*std::cout << "Source image has " << bitsPerPixel
69  << " bits per pixel. Converting to 32-bit colour." << std::endl;*/
70  bitmap32 = FreeImage_ConvertTo32Bits(bitmap);
71  }
72  // Se obtiene las dimensiones de la imagen.
73  int imageWidth = FreeImage_GetWidth(bitmap32);
74  int imageHeight = FreeImage_GetHeight(bitmap32);
75  /*std::cout << "Image: " << m_fileName << " is size: " << imageWidth << "x"
76  << imageHeight << "." << std::endl;*/
77 
78  // Se obtiene un apuntador a los datos de la textura como un arreglo de unsigned bytes.
79  GLubyte* textureData = FreeImage_GetBits(bitmap32);
80 
81  // Se genera un buffer para textura en la GPU
82  glGenTextures(1, &m_textureObj);
83  glBindTexture(m_textureTarget, m_textureObj);
84 
85  // Se envian los datos de la textura
86  glTexImage2D(GL_TEXTURE_2D, // Tipo de textura
87  0, // Niveles del Mipmap
88  GL_RGBA, //Formato intero, RGBA
89  imageWidth, // Ancho de la textura
90  imageHeight, // Ancho de la textura
91  0, // Borde de la textura
92  GL_BGRA, // Formato que se maneja la textura
93  GL_UNSIGNED_BYTE, // Tipo de datos de la textura
94  textureData); // Imagen que se usa para esta textura
95  // Se indica el tipo de interpolación para ajustar la imagen que se cargo a la GPU
96  glTexParameterf(m_textureTarget, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
97  glTexParameterf(m_textureTarget, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
98 
99  // Unload the 32-bit colour bitmap
100  // Se desecha los datos de la textura
101  FreeImage_Unload(bitmap32);
102 
103  // Si se convirtio la imagen a 32 bits por pixel eliminamos el bitmap
104  if (bitsPerPixel != 32) {
105  FreeImage_Unload(bitmap);
106  }
107 
108  // Desenlazamos la textura
109  glBindTexture(m_textureTarget, 0);
110  return true;
111 }
