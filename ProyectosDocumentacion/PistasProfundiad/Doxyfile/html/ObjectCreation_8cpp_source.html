173 bool * ObjectCreation::createCullFacesPointer(Vertex3 * vertices,
174  GLuint * index, int sizeIndex,
175  glm::vec3 camaraPosition) {
176  int sizeCullFacesPointer = sizeIndex / 3;
177  int indexFace = 0;
178  bool * cullFacesPtr = new bool[sizeCullFacesPointer];
179  for (int i = 0; i < sizeIndex; i += 3) {
180  glm::vec3 v1 = vertices[index[i + 1]].xyz - vertices[index[i]].xyz;
181  glm::vec3 v2 = vertices[index[i + 2]].xyz - vertices[index[i]].xyz;
182  glm::vec3 normal = glm::cross(v1, v2);
183  normal = glm::normalize(normal);
184  if (glm::dot(normal, camaraPosition - vertices[index[i]].xyz) > 0)
185  cullFacesPtr[indexFace] = true;
186  else
187  cullFacesPtr[indexFace] = false;
188  indexFace++;
189  }
190  return cullFacesPtr;

201  Vertex3 * vertices, GLuint * indices) {
202  // Uso local de name spaces.
203  using namespace glm;

206  const float pi = 3.1415926535897932384626433832795f;
207  const float _2pi = 2.0f * pi;

209  vector<vec3> positions;
210  vector<vec3> normals;
211  vector<vec2> textureCoords;
212 
213  for (int i = 0; i <= stacks; ++i) {
214  // Coordenadas V de la textura.
215  float V = i / (float) stacks;
216  float phi = V * pi;
217 
218  for (int j = 0; j <= slices; ++j) {
219  // Coordenadas U de la textura.

228  }
229  }

231  // Generaicón de indices.
232  for (int i = 0; i < slices * stacks + slices; ++i) {
233  *(indices + i * 6) = i;
