44 bool RayCasting::RayTriangleIntersect(glm::vec3 * orig, glm::vec3 * dir,
45  glm::vec3 *vert0, glm::vec3 * vert1,
46  glm::vec3 * vert2, double *t, double *u,
47  double *v) {
48  glm::vec3 edge1, edge2, tvec, pvec, qvec;
49  double det, inv_det;
50  // Se encuentran los vectores de dos aristas que contengan a ver0
51  edge1 = glm::vec3(*vert1 - *vert0);
52  edge2 = glm::vec3(*vert2 - *vert0);
53  // Se calcula el determiante - se usa para calcular el parametro U.
54  pvec = glm::cross(*dir, edge2);
55  // Si el determinante es muy cercano a cero, rayo cae en el plano
56  // que contiene al triángulo.
57  det = glm::dot(edge1, pvec);
58  if (det > -EPSILON && det < EPSILON)
59  return false;
60  inv_det = 1.0 / det;
61  // Distancia entre el vert0 al origen del rayo.
62  tvec = glm::vec3(*orig - *vert0);
63  // Calcula el parametro u y prueba su valor.
64  *u = glm::dot(tvec, pvec) * inv_det;
65  if (*u < 0.0 || *u > 1.0)
66  return false;
67  qvec = glm::cross(tvec, edge1);
68  // Calcula el parametro v y prueba que este dentro del rango.
69  *v = glm::dot(*dir, qvec) * inv_det;
70  if (*v < 0.0 || *u + *v > 1.0)
71  return false;
72  // Calcula t, interseccion del rayo con el triángulo
73  *t = glm::dot(edge2, qvec) * inv_det;
74  return true;
75 }

239 bool RayCasting::RaySphereIntersect(glm::vec3 * orig, glm::vec3 * dest,
240  glm::vec3 * dir, glm::vec3 * vSphereCenter,
241  float sphereRadius) {
242  // Vector del Origen del rayo al centro de la esfera.
243  glm::vec3 vDirToSphere = *vSphereCenter - *orig;
244 
245  // Distancia del origen al destino del rayo.
246  float fLineLength = glm::distance(*orig, *dest);
247 
248  // Proyección escalar de vDirToSphere sobre la direccion del rayo.
249  float t = glm::dot(vDirToSphere, *dir);
250 
251  glm::vec3 vClosestPoint;
252  // Si la distancia proyectada del origen es menor o igual que cero
253  // Significa que el punto mas cercano al centro es el origen.
254  if (t <= 0.0f)
255  vClosestPoint = *orig;
256  // Si la proyección escalar del origen es mayor a distancia del origen
257  // al destino, el punto mas cercano es el destino.
258  else if (t >= fLineLength)
259  vClosestPoint = *dest;
260  // En caso contrario de calcula el punto sobre la linea usando t.
261  else
262  vClosestPoint = *orig + *dir * (t);
263 
264  // Se pureba si el punto mas cercao esta contenido en el radio de la esfera.
265  return glm::distance(*vSphereCenter, vClosestPoint) <= sphereRadius;
266 }
