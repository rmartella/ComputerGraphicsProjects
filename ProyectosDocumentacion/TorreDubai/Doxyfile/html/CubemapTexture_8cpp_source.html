67 bool CubemapTexture::Load() {
68  glGenTextures(1, &textureObj);
69  glBindTexture(GL_TEXTURE_CUBE_MAP, textureObj);
70 
71  // Tipo de interpolación de mapeo de la textura a los trigangulos.
72  glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
73  glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_MIN_FILTER,
74  GL_NEAREST_MIPMAP_NEAREST);
75  glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
76  glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
77  glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_R, GL_CLAMP_TO_EDGE);
78 
79  // Por cada cara del skydome.
80  for (unsigned int i = 0; i < ARRAY_SIZE_IN_ELEMENTS(types); i++) {
81  // Se obtiene del nombre del archivo a cargar en la GPU como un apuntador a un arreglo de chars
82  const char* filename = fileNames[i].c_str();
83 
84  // Determina el formato de la imagen
85  FREE_IMAGE_FORMAT format = FreeImage_GetFileType(filename, 0);
86 
87  // Si la imagen no es encontrada termina el programa
88  if (format == -1) {
89  std::cout << "No se encontró la imagen: " << fileNames[i].c_str()
90  << " - Abortando." << std::endl;
91  exit(-1);
92  }
93  // Si el formato no es soportado por FreeImage termina el programa
94  if (format == FIF_UNKNOWN) {
95  std::cout
96  << "No se puede determinar el formato de imagen - validarla extensión del archivo..."
97  << std::endl;
98 
99  // Se obtiene el formato del archivo
100  format = FreeImage_GetFIFFromFilename(filename);
101 
102  // Revisa si la librería es capaz de leer el formato
103  if (!FreeImage_FIFSupportsReading(format)) {
104  std::cout << "El formato de la imagen no puede ser leeído!"
105  << std::endl;
106  exit(-1);
107  }
108  }
109  // Si es valida la imagen y puede ser leeído, se carga la imagen en un bitap
110  FIBITMAP* bitmap = FreeImage_Load(format, filename);
111  FreeImage_FlipVertical(bitmap);
112 
113  // Obtiene el número de bits por pixel de la imagen
114  int bitsPerPixel = FreeImage_GetBPP(bitmap);
115 
116  // Convierte la imagen a 32 bits (8 bits por canal).
117  FIBITMAP* bitmap32;
118  if (bitsPerPixel == 32) {
119  /*std::cout << "Source image has " << bitsPerPixel
120  << " bits per pixel. Skipping conversion." << std::endl;*/
121  bitmap32 = bitmap;
122  } else {
123  /*std::cout << "Source image has " << bitsPerPixel
124  << " bits per pixel. Converting to 32-bit colour." << std::endl;*/
125  bitmap32 = FreeImage_ConvertTo32Bits(bitmap);
126  }
127  // Se obtiene las dimensiones de la imagen.
128  int imageWidth = FreeImage_GetWidth(bitmap32);
129  int imageHeight = FreeImage_GetHeight(bitmap32);
130  /*std::cout << "Image: " << m_fileName << " is size: " << imageWidth << "x"
131  << imageHeight << "." << std::endl;*/
132 
133  // Se obtiene un apuntador a los datos de la textura como un arreglo de unsigned bytes.
134  GLubyte* textureData = FreeImage_GetBits(bitmap32);
135 
136  // Se envian los datos de la textura
137  glTexImage2D(types[i], // Tipo de textura
138  0, // Niveles del Mipmap
139  GL_RGBA, //Formato intero, RGBA
140  imageWidth, // Ancho de la textura
141  imageHeight, // Ancho de la textura
142  0, // Borde de la textura
143  GL_BGRA, // Formato que se maneja la textura
144  GL_UNSIGNED_BYTE, // Tipo de datos de la textura
145  textureData); // Imagen que se usa para esta textura
146 
147  // Unload the 32-bit colour bitmap
148  // Se desecha los datos de la textura
149  FreeImage_Unload(bitmap32);
150 
151  // Si se convirtio la imagen a 32 bits por pixel eliminamos el bitmap
152  if (bitsPerPixel != 32) {
153  FreeImage_Unload(bitmap);
154  }
155  }
156 
157  // Se genera el BitMap.
158  glGenerateMipmap(GL_TEXTURE_CUBE_MAP);
159  glBindTexture(GL_TEXTURE_CUBE_MAP, 0);
160 
161  return true;
162 }
