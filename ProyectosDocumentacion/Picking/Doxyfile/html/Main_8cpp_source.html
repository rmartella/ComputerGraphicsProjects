334 void GLApplication::SolidSphere(float radius, int slices, int stacks,
335  Vertex3 * vertices, GLuint * indices) {
336  // Uso local de name spaces.
337  using namespace glm;
338  using namespace std;
339 
340  const float pi = 3.1415926535897932384626433832795f;
341  const float _2pi = 2.0f * pi;
342 
343  vector<vec3> positions;
344  vector<vec3> normals;
345  vector<vec2> textureCoords;
346 
347  for (int i = 0; i <= stacks; ++i) {
348  // Coordenadas V de la textura.
349  float V = i / (float) stacks;
350  float phi = V * pi;
351 
352  for (int j = 0; j <= slices; ++j) {
353  // Coordenadas U de la textura.
354  float U = j / (float) slices;
355  float theta = U * _2pi;
356 
357  float X = cos(theta) * sin(phi);
358  float Y = cos(phi);
359  float Z = sin(theta) * sin(phi);
360  // Se calculan los vertices, UV, normales de la esfera.
361  *(vertices + i * (slices + 1) + j) = {vec3(X, Y, Z) * radius, vec2(U, V), vec3(X, Y, Z)};
362  }
363  }
364 
365  // Generaicón de indices.
366  for (int i = 0; i < slices * stacks + slices; ++i) {
367  *(indices + i * 6) = i;
368  *(indices + i * 6 + 1) = i + slices + 1;
369  *(indices + i * 6 + 2) = i + slices;
370  *(indices + i * 6 + 3) = i + slices + 1;
371  *(indices + i * 6 + 4) = i;
372  *(indices + i * 6 + 5) = i + 1;
373  }
374 }
