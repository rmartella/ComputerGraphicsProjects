64 bool Picking::RayTriangleIntersect(glm::vec3 orig, glm::vec3 dir,
65  glm::vec3 vert0, glm::vec3 vert1,
66  glm::vec3 vert2, double *t, double *u,
67  double *v) {
68  glm::vec3 edge1, edge2, tvec, pvec, qvec;
69  double det, inv_det;
70  // Se encuentran los vectores de dos aristas que contengan a ver0
71  edge1 = glm::vec3(vert1 - vert0);
72  edge2 = glm::vec3(vert2 - vert0);
73  // Se calcula el determiante - se usa para calcular el parametro U.
74  pvec = glm::cross(dir, edge2);
75  // Si el determinante es muy cercano a cero, rayo cae en el plano
76  // que contiene al triángulo.
77  det = glm::dot(edge1, pvec);
78  if (det > -EPSILON && det < EPSILON)
79  return false;
80  inv_det = 1.0 / det;
81  // Distancia entre el vert0 al origen del rayo.
82  tvec = glm::vec3(orig - vert0);
83  // Calcula el parametro u y prueba su valor.
84  *u = glm::dot(tvec, pvec) * inv_det;
85  if (*u < 0.0 || *u > 1.0)
86  return false;
87  qvec = glm::cross(tvec, edge1);
88  // Calcula el parametro v y prueba que este dentro del rango.
89  *v = glm::dot(dir, qvec) * inv_det;
90  if (*v < 0.0 || *u + *v > 1.0)
91  return false;
92  // Calcula t, interseccion del rayo con el triángulo
93  *t = glm::dot(edge2, qvec) * inv_det;
94  return true;
95 }

109 bool Picking::RaySphereIntersect(glm::vec3 * orig, glm::vec3 * dest,
110  glm::vec3 * dir, glm::vec3 * vSphereCenter,
111  float sphereRadius, float * t) {
112 
113  // Vector del Origen del rayo al centro de la esfera.
114  glm::vec3 vDirToSphere = *vSphereCenter - *orig;
115 
116  // Distancia del origen al destino del rayo.
117  float fLineLength = glm::distance(*orig, *dest);
118 
119  // Proyección escalar de vDirToSphere sobre la direccion del rayo.
120  *t = glm::dot(vDirToSphere, *dir);
121 
122  glm::vec3 vClosestPoint;
123  // Si la distancia proyectada del origen es menor o igual que cero
124  // Significa que el punto mas cercano al centro es el origen.
125  if (*t <= 0.0f)
126  vClosestPoint = *orig;
127  // Si la proyección escalar del origen es mayor a distancia del origen
128  // al destino, el punto mas cercano es el destino.
129  else if (*t >= fLineLength)
130  vClosestPoint = *dest;
131  // En caso contrario de calcula el punto sobre la linea usando t.
132  else
133  vClosestPoint = *orig + *dir * (*t);
134 
135  // Se pureba si el punto mas cercao esta contenido en el radio de la esfera.
136  return glm::distance(*vSphereCenter, vClosestPoint) <= sphereRadius;
137 }
